#include <stdio.h>

int main() {
    int n, sum = 0;
    printf("Nhap mot so nguyen: ");
    scanf("%d", &n);
            
    for (int i = 1; i <= n / 2; i++) {
        if (n % i == 0) {
            sum += i;
        }
    }
            
    if (sum == n && n != 0) {
        printf("%d la so hoan hao.\n", n);
    }else {
        printf("%d khong phai la so hoan hao.\n", n);
    }   
    return 0;
}
