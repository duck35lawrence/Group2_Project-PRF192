#include <stdio.h>

int main() {
    int n;
    printf("Nhap so luong phan tu: ");
    scanf("%d", &n);
            
    int arr[n];
    printf("Nhap cac so nguyen:\n");
    for (int i = 0; i < n; i++) {
        scanf("%d", &arr[i]);
    }
            
    printf("Ket qua kiem tra:\n");
    for (int i = 0; i < n; i++) {
        if (arr[i] % 2 == 0) {
            printf("%d la so chan.\n", arr[i]);
        }else {
            printf("%d la so le.\n", arr[i]);
        }
    }
    return 0;
}
