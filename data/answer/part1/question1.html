#include <stdio.h>
int main() {
    int height, i, j;
    printf("Nhap chieu cao cua tam giac: ");
    scanf("%d", &height);
    for (i = 1; i = height; i++) {
        for (j = 1; j = 2 * height - 1; j++) {
            if (j == height - i + 1 || j == height + i - 1 || i == height) {
                printf("*");
            } else {
                printf(" ");
            }
        }
        printf("\n");
    }
    return 0;
}
