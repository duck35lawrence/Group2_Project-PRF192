#include <stdio.h>
#define PI 3.14159
  
int main() {
    float banKinh, chuVi, dienTich;
  
    printf("Nhập bán kính hình tròn: ");
    scanf("%f", &banKinh);

    chuVi = 2 * PI * banKinh;
    dienTich = PI * banKinh * banKinh;
  
    printf("Chu vi hình tròn: %2f\n, chuVi);
    printf("Diện tích hình tròn: %2f\n, dienTich);
  
    return 0;
}
