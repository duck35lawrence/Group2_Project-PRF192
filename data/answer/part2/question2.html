#include <stdio.h>
    
    int main() {
        float canh, chuVi, dienTich;

        printf("Nhập chiều dài cạnh của hình vuông: ");
        scanf("%f", &canh);
    
        chuVi = 4 * canh;
        dienTich = canh * canh;
    
        printf("Chu vi hình vuông: %.2f\n", chuVi);
        printf("Diện tích hình vuông: %.2f\n", dienTich);
    
        return 0;
    }
