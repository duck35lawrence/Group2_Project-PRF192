#include <stdio.h>
#include <math.h>

    int main() {
        int a, b, c, s, chuVi, dienTich;

        printf("Nhập độ dài ba cạnh tam giác: ");
        scanf("%f %f %f", &a, &b, &c);

        chuVi = a + b + c;
        s = (a + b + c) / 2;
        dienTich = sqrt(s * (s - a) * (s - b) * (s - c));

        printf("Chu vi tam giác: %2f\n, chuVi);
        printf("Diện tích tam giác: %2f\n, dienTich);
    
        return 0;
    }
