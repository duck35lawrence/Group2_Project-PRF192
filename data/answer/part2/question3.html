#include <stdio.h>

int main() {
  float dai, rong, chuVi, dienTich;
    
  printf("Nhập chiều dài của hình chữ nhật: ");
  scanf("%f", &dai);
  printf("Nhập chiều rộng của hình chữ nhật: ");
  scanf("%f", &rong);
    
  chuVi = 2 * (dai + rong);
  dienTich = dai * rong;
    
  printf("Chu vi hình chữ nhật: %.2f\n", chuVi);
  printf("Diện tích hình chữ nhật: %.2f\n", dienTich);
    
  return 0;
}
